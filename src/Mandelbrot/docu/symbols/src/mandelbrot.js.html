 24 /**
 25  * @description Clase que representa un numero complejo, que se caracteriza
 26  * por tener una parte real y una parte imaginaria.
 27  */
 28 class ComplexNumber {
 29   constructor(realPart, complexPart) {
 30     this.realPart_ = realPart;
 31     this.complexPart_ = complexPart;
 32   }
 33   // ----------Setters y getters-------------
 34 
 35   set realPart (value) {
 36     this.realPart_ = value;
 37   }
 38 
 39   set complexPart(value) {
 40     this.complexPart_ = value;
 41   }
 42 
 43   get realPart() {
 44     this.realPart_;
 45   }
 46 
 47   get complexPart() {
 48     this.complexPart_;
 49   }
 50 
 51   // ----------Operaciones de números complejos-----------
 52 
 53   add(toAdd) {
 54     let addedReal = this.realPart_ + toAdd.realPart_;
 55     let addedComplex = this.complexPart_ + toAdd.complexPart_;
 56     return new ComplexNumber(addedReal, addedComplex);
 57   }
 58 
 59   square() {
 60     let squaredReal = this.realPart_ * this.realPart_ 
 61                       - this.complexPart_ * this.complexPart_;
 62     let squaredComplex = 2 * this.realPart_ * this.complexPart_;
 63     return new ComplexNumber(squaredReal, squaredComplex);
 64   }
 65 
 66   module() {
 67     return Math.round((Math.sqrt(Math.pow(this.realPart_, 2)
 68      + Math.pow(this.complexPart_, 2))) * 1000) / 1000;
 69   }
 70 }

 74 /**
 75  * @description Función que dibuja en cada punto del canvas un punto, cuyo color
 76  * variará en función de si está en el conjunto de mandelbrot o no.
 77  * Este color se obtiene del número de iteraciones calculado por el algoritmo de
 78  * Mandelbrot.
 79  * @param {Object} CTX Contexto del canvas
 80  * @param {*} CANVAS Identificador del canvas
 81  */
 82 function drawMandelbrotOnCanvas(CTX, CANVAS) {
 83  let toScale = 280;
 84  let xDisp = 1.5;
 85  let yDisp = 1.05;
 86  let iterationsNumber = 0;
 87 
 88  for (let x = 0; x < CANVAS.width; x++) {
 89    for (let y = 0; y < CANVAS.height; y++) {
 90      let complex = new ComplexNumber(x / toScale - xDisp,
 91                                      y / toScale - yDisp);                             
 92      iterationsNumber = mandelbrotAlgorithm(complex);
 93      drawPoint(CTX, iterationsNumber, x, y);
 94    }
 95  }
 96 }

 99 function drawPoint(CTX, iterationsNumber, px, py) {
100   const lightness = ((iterationsNumber < MAX_ITER) ? iterationsNumber / MAX_ITER * 100 + 15 : 0);
101   const hue = (210 * iterationsNumber / MAX_ITER);
102  
103   CTX.fillStyle = 'hsl(' + hue +', 100%,' + lightness + '%)';
104   CTX.fillRect(px, py, 1, 1);
105 }

107 /**
108  * Función que itera el calculo de Mandelbrot hasta que el número sea mayor
109  * que un umbral o se den un número n de iteraciones.
110  * @param {Object} complexNumber Numero complejo
111  * @returns Número de iteraciones
112  */
113 function mandelbrotAlgorithm(complexNumber) {
114   let zModule;
115   let zComplex = complexNumber;
116   let currentIteration = 0;
117   while((currentIteration < MAX_ITER) && (!isGreaterThan(zModule, 2))) {
118     // z = z^2 + c
119     let zComplexSquared = zComplex.square(); // z^2
120     zComplex = zComplexSquared.add(complexNumber); // z = z^2 + c[i]
121     currentIteration++;
122     zModule = zComplex.module();
123     // console.log("|z| = ", zModule);
124   }
125   return currentIteration;
126 }
127 
128 /**
129  * Función que itera el algoritmo de Mandelbrot para comprobar si un número
130  * complejo c se encuentra en el conjunto. Para ello, se tiene que dar que
131  * se cumplan las n iteraciones (10000).
132  * @param {Number} complexOnInterval Número complejo que se valora.
133  * @returns True si no está en el conjunto, False si no está.
134  */
135 function isOnMandelbrotSet(complexOnInterval) {
136   let zModule;
137   let zComplex = complexOnInterval;
138   let currentIteration = 0;
139   while((currentIteration < MAX_ITER)) {
140     let zComplexSquared = zComplex.square(); // z^2
141     zComplex = zComplexSquared.add(complexOnInterval); // z = z^2 + c[i]
142     currentIteration++;
143     zModule = zComplex.module();
144  
145     if (isGreaterThan(zModule, 2))
146       return true;
147   }
148   return false;
149 }

151 /**
152  * Funcion que comprueba si un número es mayor que otro
153  * @param {Number} absComplexNumber Módulo de un número complejo
154  * @param {Number} umbral Valor umbral que se compara
155  * @returns True si es mayor que el umbral, False en otro caso.
156  */
157 function isGreaterThan(absComplexNumber, umbral) {
158   if (absComplexNumber > umbral)
159     return true;
160   return false;
161 }

163 /**
164  * @description Función que se encarga de generar números aleatorios en un
165  * intervalo dado.
166  * @param {Object} from Punto delimitador izquierdo
167  * @param {Object} to Punto delimitador derecho
168  * @returns Array que contiene todos los puntos generados.
169  */
170 function generatePointsBetween(fromR, toR, fromI, toI) {
171   let real = Math.random() * (toR - fromR) + fromR;
172   let complex = Math.random() * (toI - fromI) + fromI;
173   return new ComplexNumber(real, complex);
174 }

176 /**
177  * @description Método que lleva a cabo la representación de Monte Carlo
178  *              del Conjunto de Mandelbrot.
179  *              Esto es, dado un intervalo de números complejos, generar
180  *              N números complejos aleatorios, aplicarles el algoritmo
181  *              de Mandelbrot para comprobar si se encuentran dentro de
182  *              dicho conjunto y calcular el área del conjunto.
183  * @param {Number} complexNumberContainer Numero de números complejos que se
184  *                                        generan aleatoriamente.
185  */
186 function monteCarloRepresentation(complexNumberContainer, CTX) {
187   let nInside = 0;
188   const RE_FIRST = -2;
189   const RE_SECOND = 0.5;
190   const IM_FIRST = 0;
191   const IM_SECOND = 1.125;
192   for (let currentComplex = 0; currentComplex < complexNumberContainer; currentComplex++) {
193     let randomComplexPoint = generatePointsBetween(RE_FIRST, RE_SECOND, IM_FIRST, IM_SECOND);
194     if (!isOnMandelbrotSet(randomComplexPoint))
195       nInside++;
196   }
197   calculateAreaAndError(complexNumberContainer, nInside, CTX);   
198 }
199 
200 /**
201  * @description Función que calcula el área y el error del conjunto de
202  * Mandelbrot e imprime sus valores por consola.
203  * @param {number} nPoints Número de puntos generados en el contenedor
204  * @param {number} nInside Número de puntos que entraron dentro de la 
205  * representación de Monte Carlo 
206  */
207 function calculateAreaAndError(nPoints, nInside, CTX) {
208   
209   const AREA = 2 * 2.5 * 1.125 * nInside / nPoints;
210   const ERROR = AREA / Math.sqrt(nPoints);
211 
212   printTextOnCanvas(CTX, "Area: ", AREA, 40, 40);
213   printTextOnCanvas(CTX, "Error: ", ERROR, 40, 60);
214 
215 }

217 /**
218  * @description Función que se encarga de imprimir en el canvas un texto dado.
219  * @param {Object} CTX Contexto del canvas
220  * @param {String} text Texto descriptivo de lo que se quiere imprimir
221  * @param {Number} value Valor que se imprime
222  * @param {*} px Coordenada X del punto que se quiere imprimir
223  * @param {*} py Coordenada Y del punto que se quiere imprimir
224  */
225 function printTextOnCanvas(CTX, text, value, px, py) {
226   CTX.font = 'bold 15pt Calibri';
227   CTX.fillStyle = "black";
228   CTX.fillText(`${text} ${value}`,px, py);
229 }
