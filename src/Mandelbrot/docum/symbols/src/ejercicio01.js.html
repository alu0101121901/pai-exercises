 16 /**
 17  * @description Función que invoca 1 a 1 las formas necesarias
 18  * para representar la señal de Prohibido el paso.
 19  * @param {Object} CTX Contexto del canvas
 20  * @param {Object} CANVAS Identificador del canvas
 21  */
 22 function drawSign(CTX, CANVAS) {
 23   // Circulo rojo
 24   let radiusRedCircle = CANVAS.width * 0.40
 25   drawCircle(CTX, CANVAS.width / 2, CANVAS.height / 2, radiusRedCircle, 0, Math.PI*2, "red");
 26   // Linea blanca
 27   drawLine(CTX, CANVAS.width / 2, CANVAS.height / 2, radiusRedCircle - 10, 0, Math.PI*2);
 28   // Rectángulo del centro
 29   drawRect(CTX, CANVAS.width / 4, CANVAS.height / 2 * 0.9, CANVAS);
 30 }

 32 /**
 33  * @description
 34  * @param {Object} CTX Contexto del canvas
 35  * @param {Number} px Coordenada x
 36  * @param {Number} py Coordenada y
 37  * @param {Object} CANVAS Identificador del canvas
 38  */
 39 function drawRect(CTX, px, py, CANVAS) {
 40   CTX.beginPath();
 41   CTX.fillStyle = 'white';
 42   CTX.rect(px, py, CANVAS.width * 0.5 , CANVAS.height * 0.12, 15);
 43   CTX.stroke();
 44   CTX.fill();
 45 }

 51 /**
 52  * @description Función que se encarga de dibujar la línea blanca
 53  * de la señal de prohibido el paso.
 54  * @param {Object} CTX Contexto del canvas
 55  * @param {Number} px Coordenada x
 56  * @param {Number} py Coordenada y
 57  * @param {Number} radius Radio del círculo
 58  * @param {Number} start Punto desde el que se empieza a trazar
 59  * el arco.
 60  * @param {Number} end Punto en el que se termina de trazar el arco.
 61  */
 62 function drawLine(CTX, px, py, radius, start, end) {
 63   CTX.beginPath();
 64   CTX.arc(px, py, radius, start, end);
 65   CTX.lineWidth = 5;
 66   CTX.strokeStyle = '#FFFFFF';
 67   CTX.stroke();
 68 }
 69 
 70 /**
 71  * @description Función que dibuja el círculo rojo de la señal de prohibido
 72  * el paso.
 73  * @param {Object} CTX Contexto del canvas
 74  * @param {Number} px Coordenada x
 75  * @param {Number} py Coordenada y
 76  * @param {Number} radius Radio del círculo
 77  * @param {Number} start Punto desde el que se empieza a trazar
 78  * el arco.
 79  * @param {Number} end Punto en el que se termina de trazar el arco.
 80  * @param {String} color Color con el que se quiere dibujar el círculo
 81  */
 82 function drawCircle(CTX, px, py, radius, start, end, color) {
 83   CTX.beginPath();
 84   CTX.arc(px, py, radius, start, end);
 85   CTX.fillStyle = `${color}`;
 86   CTX.fill();
 87   CTX.stroke();
 88 }
